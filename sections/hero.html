const studio: ProductStudio = {
  name: "Tesar Pratama",
  focus: [
    "Sistem Billing PlayStation",

  build: async (requirements) => {
    const architecture = designSystem(requirements);
    const product = await craftProduct(architecture);

    return deploy(product, {
      quality: "premium",
      scale: "business-ready"
    });
  }
};
